# 三角形

![](/images/三角形/QQ截图20220329232219.png)

思路：由于最大n就到100，三重循环暴力枚举三个点即可

```cpp
#include <iostream>
using namespace std;
#include <vector>

typedef long long LL;
#define x first
#define y second
typedef pair<int,int> PII;
vector<PII> a;

int main()
{
    int n;
    cin >> n;
    while (n -- )
    {
        int x,y;
        cin >> x >> y;
        a.push_back({x, y});
    }
    
    int ans = 0;
    for(int i = 0;i < a.size();i++)
    {
        for(int j = i + 1;j < a.size();j++)
        {
            for(int k = j + 1;k < a.size();k++)
            {
                int x1 = a[i].x, y1 = a[i].y;
                int x2 = a[j].x, y2 = a[j].y;
                int x3 = a[k].x, y3 = a[k].y;

                if(x1 == x2 && y1 == y3)
                {
                    ans = max(ans, abs(y2 - y1) * abs(x1 - x3));
                }
                else if(x1 == x3 && y1 == y2)
                {
                    ans = max(ans, abs(y1 - y3) * abs(x1 - x2));
                }
                else if(x2 == x1 && y2 == y3)
                {
                    ans = max(ans, abs(y2 - y1) * abs(x3 - x2));
                }
                else if(x2 == x3 && y2 == y1)
                {
                    ans = max(ans, abs(y2 - y3) * abs(x2 - x1));
                }
                else if(x3 == x1 && y3 == y2)
                {
                    ans = max(ans, abs(y1 - y3) * abs(x3 - x2));
                }
                else if(x3 == x2 && y3 == y1)
                {
                    ans = max(ans, abs(y2 - y3) * abs(x1 - x3));
                }
            }
        }
    }
    
    cout << ans << endl;
    
    return 0;
}
```
